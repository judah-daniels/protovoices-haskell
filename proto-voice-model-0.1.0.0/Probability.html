multinomialLogProb :: Vector Double -> Vector Double -> Double Source #

Computes the multinomial log probability density function parameterised by prob, evaluated at the vector xs. The sum of the observations for all categories must add up to the number of samples, n. The probabilties must be NORMALISED!

data Prob a Source #

Sample n elements from a list sample :: Show a => Int -> [a] -> Reservoir a [a] sample n xs = do let len = length xs let updateReservoir x = do ys <- get i <- lift $ randomRIO (0, len) if i < n then put (take i ys ++ [x] ++ drop (i + 1) ys) else return () liftIO $ putStrLn "Reservoir sampling..." liftIO $ putStrLn $ "List length: " ++ show len liftIO $ putStrLn $ "Reservoir size: " ++ show n liftIO $ putStrLn "Reservoir:" liftIO $ putStrLn "---------" put (take n xs) forM_ (drop n xs) $ x -> do updateReservoir x get >>= liftIO . putStrLn . show get
